Menampilkan Gambar

Package dasar EBImage memiliki fungsi yaitu membaca, menulis, dan menampilkan gambar. EBImage membaca gambar dengan menggunakan function readImage() dimana dalam function tersebut nantinya dimasukkan url gambar yang ingin ditampilkan.

Package EBImage saat ini mendukung 3 jenis format gambar, yaitu jpg, png, dan tiff.

library("EBImage")
img <- readImage("dandelion.jpg")
# Menampilkan gambar di penampil JavaScript pada tab browser web
display(img, method="browser")
# Menampilkan gambar seperti biasa tetapi dengan tambahan label di atasnya
display(img, method="raster")

Filter Gambar

Pra-pemrosesan gambar yang melibatkan pembersihan gambar dengan menghilangkan noise melalui penghalusan. Cara kerjanya adalah dengan menentukan jendela untuk unkuran yang dipilih pada setiap pixel dan rata-rata nilainya. Setelah menerapkan prosedur tersebut, gambar baru yang dihaluskan akan diperoleh.

Fourier Transform

FFT Shift Transform

fftshift <- function(img_ff, dim = -1) {

  rows <- dim(img_ff)[1]    
  cols <- dim(img_ff)[2]
  # planes <- dim(img_ff)[3]

  swap_up_down <- function(img_ff) {
    rows_half <- ceiling(rows/2)
    return(rbind(img_ff[((rows_half+1):rows), (1:cols)], img_ff[(1:rows_half), (1:cols)]))
  }

  swap_left_right <- function(img_ff) {
    cols_half <- ceiling(cols/2)
    return(cbind(img_ff[1:rows, ((cols_half+1):cols)], img_ff[1:rows, 1:cols_half]))
  }
  
  if (dim == -1) {
    img_ff <- swap_up_down(img_ff)
    return(swap_left_right(img_ff))
  }
  else if (dim == 1) {
    return(swap_up_down(img_ff))
  }
  else if (dim == 2) {
    return(swap_left_right(img_ff))
  }
  else if (dim == 3) {
  # Menggunakan package 'abind'
  # install.packages("abind")
    library(abind)
    
    planes <- dim(img_ff)[3]
    rows_half <- ceiling(rows/2)
    cols_half <- ceiling(cols/2)
    planes_half <- ceiling(planes/2)
    
    img_ff <- abind(img_ff[((rows_half+1):rows), (1:cols), (1:planes)], 
                    img_ff[(1:rows_half), (1:cols), (1:planes)], along=1)
    img_ff <- abind(img_ff[1:rows, ((cols_half+1):cols), (1:planes)], 
                    img_ff[1:rows, 1:cols_half, (1:planes)], along=2)
    img_ff <- abind(img_ff[1:rows, 1:cols, ((planes_half+1):planes)], 
                    img_ff[1:rows, 1:cols, 1:planes_half], along=3)
    return(img_ff)
  }
  else {
    stop("Invalid dimension parameter")
  }
}

library(EBImage)
img2 <- readImage("sunflower.jpg")
display(img2, method="raster", all=TRUE)

# Intensity Histogram
hist(img2)

# Generate negative image
img_neg <- max(img2) - img2
display( img_neg )

# crop and threshold
img_crop <- img2  # [200:578, 250:750, ]
img_thresh <- img_crop > 0.55
display(img_thresh)

# Low-pass Filter/smooth
w <- makeBrush(size = 31, shape = 'gaussian', sigma = 5)
img_flo <- filter2(img2, w)
display(img_flo)

# High-pass filtering Laplace Filter/Edge detection
fhi <- matrix(1, nrow = 3, ncol = 3)
fhi[2, 2] <- -7
img_fhi <- filter2(img2, fhi)
display(img_fhi)

# Denoising
img_median <- medianFilter(img2, size=5, cacheSize=20000)
display(img_median)

# Segmentation
grayimage <- channel(img2,"gray")
threshold <- otsu(grayimage)
img2_th <- combine( mapply(function(frame, th) frame > th, 
                         getFrames(img2), 
                         threshold, SIMPLIFY=FALSE) )
display(img2_th, all=TRUE)

disc <- makeBrush(51, "disc")
disc <- disc/sum(disc)
offset <- 0.0001
img2_bg <- filter2(img2, disc)
img2_th <- img2 < img2_bg + offset
display(img2_th, all=TRUE)

# Watershed segmentation
nmask <- watershed(distmap(img2_th), tolerance=5, ext=20)
display(colorLabels(nmask), all=TRUE)

# Voronoi image tesselation
voronoiExamp <- propagate(x = nmask, seeds = nmask, lambda = 0.001)
voronoiPaint <- colorLabels (voronoiExamp)
display(voronoiPaint, all=TRUE)

im2 <- rgbImage(green=2.0*img2_bg, blue=img_median, red=voronoiPaint[ , , , 1])
display(im2, all = TRUE)

ctmask = opening(img_fhi<0.6, makeBrush(5, shape='disc'))
cmask = propagate(img_fhi, seeds=nmask, mask=ctmask)

segmented = paintObjects(channel(cmask,"gray"), img2, col='#ff00ff')
